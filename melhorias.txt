```python
import difflib
import json
import re
import traceback
from typing import Dict, Any, List, Optional, Callable
from src.tools.base import Tool

class ToolValidator:
    def __init__(self, llm_client: Optional[Callable] = None, cwd: str = "."):
        self.llm_client = llm_client
        self.cwd = cwd

    def _heuristic_repair(self, tool_call: Dict[str, Any], tools: Dict[str, Tool], 
                          last_user_message: str = "") -> Optional[Dict[str, Any]]:
        """Tenta corrigir chamadas com heurísticas ANTES de chamar o LLM."""
        tool_name = tool_call.get("name")
        if not tool_name or tool_name not in tools:
            return None
        
        args = tool_call.get("arguments", {})
        tool = tools[tool_name]
        required = getattr(tool, "required_args", [])
        
        # Se faltar 'path', usa o diretório atual ou extrai da última mensagem
        if "path" in required and not args.get("path"):
            # Tenta extrair caminho da última mensagem do usuário
            path_match = re.search(r'["\']([^"\']+)["\']', last_user_message)
            if path_match:
                args["path"] = path_match.group(1)
            else:
                # Se for comando de listagem, usa o cwd
                if tool_name in ["list_directory", "read_file"]:
                    args["path"] = self.cwd
                else:
                    return None
        
        # Se faltar 'query', usa a última mensagem do usuário
        if "query" in required and not args.get("query"):
            if last_user_message:
                # Remove comandos comuns da query
                clean_query = re.sub(r'(pesquise|busque|procure|search|find|por|for)\s+', '', last_user_message, flags=re.I)
                if clean_query:
                    args["query"] = clean_query
                else:
                    return None
            else:
                return None
        
        # Se faltar 'content' (para escrita), tenta extrair
        if "content" in required and not args.get("content"):
            if last_user_message:
                # Tenta extrair algo entre aspas ou depois de "escreva"
                content_match = re.search(r'(?:escreva|write|salve|save)\s+["\']?([^"\']+)["\']?', last_user_message, re.I)
                if content_match:
                    args["content"] = content_match.group(1)
                else:
                    return None
            else:
                return None
        
        tool_call["arguments"] = args
        return tool_call

    def validate(self, tool_call: Dict[str, Any], tools: Dict[str, Tool]) -> Dict[str, Any]:
        if "name" not in tool_call or "arguments" not in tool_call:
            raise ValueError("Tool call malformada: faltando 'name' ou 'arguments'")
        
        tool_name = tool_call["name"]
        if tool_name not in tools:
            matches = difflib.get_close_matches(tool_name, tools.keys(), n=1, cutoff=0.6)
            if matches:
                suggested = matches[0]
                raise ValueError(f"Ferramenta '{tool_name}' não encontrada. Você quis dizer '{suggested}'?")
            raise ValueError(f"Ferramenta '{tool_name}' não existe.")
        
        tool = tools[tool_name]
        required_args = getattr(tool, "required_args", [])
        args = tool_call.get("arguments", {})
        
        missing = [arg for arg in required_args if arg not in args or args[arg] in (None, "")]
        if missing:
            raise ValueError(f"Argumentos obrigatórios faltando: {missing}. Recebido: {args}")
        
        return tool_call

    def repair(self, tool_call: Dict[str, Any], tools: Dict[str, Tool], 
               conversation: List[Dict], last_user_message: str = "") -> Optional[Dict[str, Any]]:
        # 1. Tenta heurística primeiro
        heuristic = self._heuristic_repair(tool_call, tools, last_user_message)
        if heuristic:
            try:
                return self.validate(heuristic, tools)
            except ValueError:
                pass  # Se a heurística falhar na validação, segue para LLM
        
        # 2. Fallback para LLM (se disponível)
        try:
            return self.validate(tool_call, tools)
        except ValueError as e:
            if not self.llm_client:
                return None
            
            tool_name = tool_call.get("name", "unknown")
            tool_schemas = {}
            for name, t in tools.items():
                tool_schemas[name] = {
                    "description": t.description,
                    "required_args": getattr(t, "required_args", [])
                }
            
            repair_prompt = f"""
            Você é um assistente que corrige chamadas de ferramentas (tool calls).
            
            A ferramenta '{tool_name}' falhou com o erro: {str(e)}
            
            Ferramentas disponíveis e seus schemas:
            {json.dumps(tool_schemas, indent=2)}
            
            A tool call original foi: {json.dumps(tool_call, indent=2)}
            
            Última mensagem do usuário: {last_user_message}
            
            Corrija a tool call. Responda APENAS com um JSON válido contendo "name" e "arguments".
            Se não for possível corrigir, responda com {{"error": "motivo"}}.
            """
            
            try:
                if hasattr(self.llm_client, "complete"):
                    response = self.llm_client.complete(repair_prompt)
                else:
                    response = self.llm_client(repair_prompt)
                
                json_match = re.search(r'\{.*\}', response, re.DOTALL)
                if not json_match:
                    return None
                
                repaired = json.loads(json_match.group())
                if "error" in repaired:
                    return None
                
                return self.validate(repaired, tools)
            except Exception:
                return None
```

```python
import re
import os
from typing import List, Dict, Any

class FallbackParser:
    def __init__(self, cwd: str = "."):
        self.cwd = cwd

    def parse(self, user_input: str) -> List[Dict[str, Any]]:
        input_lower = user_input.lower().strip()
        calls = []

        # 1. Leitura de arquivo (mais variações)
        read_patterns = [
            r'(?:leia|read|mostre|exiba|cat|show me|open|abrir?)\s+(?:o\s+)?(?:arquivo|file|conteúdo de)?\s*["\']?([^"\']+)["\']?',
            r'(?:conteúdo|content)\s+(?:do|da|de)?\s*["\']?([^"\']+)["\']?'
        ]
        for pattern in read_patterns:
            match = re.search(pattern, input_lower)
            if match:
                filepath = match.group(1).strip()
                calls.append({"name": "read_file", "arguments": {"path": filepath}})
                return calls

        # 2. Escrita em arquivo
        write_match = re.search(r'(?:escreva|write|salve|save|crie|create)\s+(.+?)\s+(?:no|em|para|no arquivo|em arquivo|in file|no caminho|em caminho)\s+["\']?([^"\']+)["\']?', input_lower)
        if write_match:
            content = write_match.group(1).strip()
            filepath = write_match.group(2).strip()
            calls.append({"name": "write_file", "arguments": {"path": filepath, "content": content}})
            return calls

        # 3. Listar diretório
        if re.search(r'(?:liste|list|ls|dir|mostre arquivos|show files|arquivos em|pastas em|o que tem em|tree)', input_lower):
            path_match = re.search(r'(?:em|in|de|do|da|no|na|pasta|diretório|directory)\s+["\']?([^"\']+)["\']?', input_lower)
            path = path_match.group(1) if path_match else self.cwd
            calls.append({"name": "list_directory", "arguments": {"path": path}})
            return calls

        # 4. Comando shell
        shell_match = re.search(r'(?:execute|run|exec|rode|comando|terminal)\s+(.+?)(?:$|,|\.)', input_lower)
        if shell_match:
            command = shell_match.group(1).strip()
            calls.append({"name": "shell", "arguments": {"command": command}})
            return calls

        # 5. Busca por código
        search_match = re.search(r'(?:busque|search|procure|find|localize)\s+(?:por|for)?\s+["\']?([^"\']+)["\']?', input_lower)
        if search_match:
            query = search_match.group(1).strip()
            calls.append({"name": "search_code", "arguments": {"query": query}})
            return calls

        # 6. Web search
        web_match = re.search(r'(?:pesquise|web search|busque na web|google|pesquisa na internet)\s+["\']?([^"\']+)["\']?', input_lower)
        if web_match:
            query = web_match.group(1).strip()
            calls.append({"name": "web_search_fetch", "arguments": {"query": query}})
            return calls

        # 7. Ajuda/exploração
        if re.search(r'(?:explore|explorar|analise|analyze|mapeie|map|investigue|investigar)', input_lower):
            path_match = re.search(r'(?:o\s+)?(?:projeto|project|diretório|pasta|directory|repo)?\s*["\']?([^"\']+)["\']?', input_lower)
            path = path_match.group(1) if path_match else self.cwd
            # O comando /explore será tratado pelo sistema, mas aqui mapeamos para listagem + busca
            calls.append({"name": "list_directory", "arguments": {"path": path}})
            return calls

        return []
```

```python
import json
from typing import List, Dict, Any
from src.llm import stream_llm  # ajuste o import conforme seu arquivo

async def summarize_conversation(messages: List[Dict[str, Any]]) -> str:
    """Resume uma lista de mensagens usando o LLM."""
    if not messages:
        return "Nenhuma conversa anterior."
    
    # Converte para texto
    conversation_text = ""
    for msg in messages[-20:]:  # Pega as últimas 20 para não estourar o contexto
        role = msg.get("role", "unknown")
        content = msg.get("content", "")
        reasoning = msg.get("reasoning", "")
        if reasoning:
            content += f" (Raciocínio: {reasoning})"
        conversation_text += f"[{role}]: {content}\n"
    
    prompt = f"""
    Você é um assistente que resume conversas de um agente de IA.
    Resuma os seguintes pontos principais da conversa abaixo, mantendo informações críticas como:
    - Arquivos que foram lidos/escritos.
    - Comandos executados.
    - Decisões importantes tomadas.
    - Erros encontrados e como foram resolvidos.
    
    ### CONVERSA:
    {conversation_text}
    
    ### RESUMO (máximo 300 palavras):
    """
    
    try:
        # Usa o mesmo cliente LLM, mas sem ferramentas
        response, _, _ = await stream_llm(
            messages=[{"role": "user", "content": prompt}],
            model="gemma-2b-it",  # ou o modelo padrão
            temperature=0.5,
            stream=False
        )
        return response.strip() if response else "Não foi possível gerar resumo."
    except Exception as e:
        return f"Erro ao sumarizar: {str(e)}"

async def check_and_summarize(session, max_tokens=4000):
    """Verifica o tamanho do histórico e sumariza se necessário."""
    total_chars = 0
    for msg in session.messages:
        total_chars += len(msg.get("content", "")) + len(msg.get("reasoning", ""))
    estimated_tokens = total_chars // 4
    
    if estimated_tokens > max_tokens:
        print("[Sistema] Contexto grande, aplicando sumarização...")
        keep_count = 5
        keep_messages = session.messages[-keep_count:]
        old_messages = session.messages[:-keep_count]
        
        summary = await summarize_conversation(old_messages)
        
        session.messages = [
            {"role": "system", "content": f"Resumo da conversa anterior: {summary}"}
        ] + keep_messages
        session._save()
        print("[Sistema] Sumarização concluída.")
```

```python
    """Busca na web, acessa os top N resultados e retorna o conteúdo limpo (versão síncrona)."""

        super().__init__(
            name="web_search_fetch",
            description="Busca na web, acessa os 3 primeiros resultados e retorna o conteúdo extraído (texto limpo).",
            required_args=["query"]
        )

    def execute(self, query: str, max_results: int = 3) -> ToolResult:  # <-- SEM async
        try:
            with DDGS() as ddgs:
                results = list(ddgs.text(query, max_results=max_results))
            
            if not results:
                return ToolResult(success=False, error="Nenhum resultado encontrado.")
            
            combined_text = f"## Resultados para: {query}\n\n"
            
            # Usando httpx síncrono (Client, não AsyncClient)
            with httpx.Client(timeout=15.0) as client:
                for i, res in enumerate(results):
                    url = res.get("href")
                    title = res.get("title", "Sem título")
                    snippet = res.get("body", "")
                    combined_text += f"### {i+1}. {title}\nFonte: {url}\n\n"
                    
                    try:
                        resp = client.get(url, follow_redirects=True)
                        if resp.status_code == 200:
                            try:
                                soup = BeautifulSoup(resp.text, 'html.parser')
                                for script in soup(["script", "style"]):
                                    script.decompose()
                                text = soup.get_text(separator=" ", strip=True)
                                text = re.sub(r'\s+', ' ', text)
                                full_content = text[:2000] + "..." if len(text) > 2000 else text
                            except:
                                text = re.sub(r'<[^>]+>', ' ', resp.text)
                                text = re.sub(r'\s+', ' ', text)
                                full_content = text[:2000] + "..." if len(text) > 2000 else text
                            
                            combined_text += f"**Conteúdo extraído:**\n{full_content}\n\n"
                        else:
                            combined_text += f"*(Não foi possível carregar a página - status {resp.status_code})*\n\n"
                    except Exception as e:
                        combined_text += f"*(Erro ao buscar conteúdo: {str(e)})*\n\n"
                    
                    if "Conteúdo extraído" not in combined_text[-200:]:
                        combined_text += f"**Snippet:**\n{snippet}\n\n"
            
            return ToolResult(success=True, output=combined_text)
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

```python
import json
import os
from collections import defaultdict
from typing import Dict, List

class MetricsAnalyzer:
    def __init__(self, log_dir: str = "logs"):
        self.log_dir = log_dir

    def analyze_stats(self) -> Dict:
        """Lê o arquivo stats.jsonl e retorna métricas agregadas."""
        stats_file = os.path.join(self.log_dir, "stats.jsonl")
        if not os.path.exists(stats_file):
            return {"error": "Arquivo de estatísticas não encontrado."}
        
        metrics = defaultdict(list)
        with open(stats_file, "r", encoding="utf-8") as f:
            for line in f:
                if not line.strip():
                    continue
                try:
                    entry = json.loads(line)
                    metric = entry.get("metric")
                    value = entry.get("value")
                    if metric:
                        metrics[metric].append(value)
                except:
                    continue
        
        # Calcula taxas
        report = {}
        for key, values in metrics.items():
            if key == "tool_repaired":
                success = sum(1 for v in values if v is True)
                total = len(values)
                report["taxa_reparo"] = f"{success}/{total} ({success/total*100:.1f}%)" if total > 0 else "0"
            elif key == "fallback_triggered":
                success = sum(1 for v in values if v is True)
                total = len(values)
                report["taxa_fallback"] = f"{success}/{total} ({success/total*100:.1f}%)" if total > 0 else "0"
        
        # Ferramentas mais usadas (dos logs de turno)
        turn_files = [f for f in os.listdir(self.log_dir) if f.startswith("session_") and f.endswith(".jsonl")]
        tool_usage = defaultdict(int)
        for tf in turn_files[-5:]:  # últimos 5 arquivos
            with open(os.path.join(self.log_dir, tf), "r", encoding="utf-8") as f:
                for line in f:
                    try:
                        data = json.loads(line)
                        tc = data.get("tool_call", {})
                        if tc:
                            tool_usage[tc.get("name", "unknown")] += 1
                    except:
                        continue
        
        report["ferramentas_mais_usadas"] = dict(sorted(tool_usage.items(), key=lambda x: x[1], reverse=True)[:5])
        return report

    def generate_prompt_boost(self) -> str:
        """Gera sugestões para melhorar o system prompt com base nos logs."""
        report = self.analyze_stats()
        if "error" in report:
            return "Sem dados suficientes para análise."
        
        suggestions = []
        # Se taxa de fallback é alta, sugerir adicionar exemplos no prompt
        if "taxa_fallback" in report:
            taxa = report["taxa_fallback"]
            try:
                perc = float(taxa.split("(")[1].replace("%)", ""))
                if perc > 30:
                    suggestions.append("- Alto uso de fallback (>30%). Adicione mais exemplos de tool calls no system prompt.")
            except:
                pass
        
        if "taxa_reparo" in report:
            try:
                perc = float(report["taxa_reparo"].split("(")[1].replace("%)", ""))
                if perc < 50:
                    suggestions.append("- Baixa taxa de reparo automático (<50%). Considere usar um modelo maior para validação ou aumentar a heurística.")
            except:
                pass
        
        if not suggestions:
            suggestions.append("- Sistema parece estável. Continue monitorando.")
        
        return "\n".join(suggestions)
```
